* vdiff
#+begin_src elisp
  (use-package vdiff
    :straight t
    :commands (vdiff-mode)
    :hook (vdiff-mode . (lambda ()
                          (define-key vdiff-mode-map (kbd "C-c") vdiff-mode-prefix-map))))
#+end_src
